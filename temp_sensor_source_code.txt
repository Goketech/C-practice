const int temp_pin = A0;
const int baselineTemp = 60;
const int BUZZER_PIN  = 8; 

void setup()
{
  Serial.begin(9600);
  pinMode (3, OUTPUT);
  
}
  
void loop()
{ 
  
  float voltage;
 float temp_Celsius;
  
  //read voltage and convert it into temperature
  /*analog read returns the sensor voltage as an integer
  /in the range of 0 to 1023*/
  voltage = analogRead(temp_pin) * 5.0 / 1023; 
  temp_Celsius=100 * voltage -50;
  
  Serial.print(" Voltage ");
  Serial.print(voltage); 
  Serial.print(" Temperature in Degrees Celsius: ");
  Serial.println(temp_Celsius);
  if (temp_Celsius < baselineTemp) {
    digitalWrite(3, LOW);
    digitalWrite(BUZZER_PIN, HIGH);
  } 
  
  if (temp_Celsius > baselineTemp) {
    digitalWrite(3, HIGH);
    digitalWrite(BUZZER_PIN, HIGH);
  }

  delay(1000);
}
